## Custom UI

### Sign Up Flow

You can use the `emailPasswordSignUp` function after collecting information from the user.

```tsx
import ThirdPartyEmailPassword from "supertokens-web-js/recipe/thirdpartyemailpassword";

async function signUp(email: string, password: string) {
    const signUpResponse = await ThirdPartyEmailPassword.emailPasswordSignUp({
        formFields: [
            {
                id: "email",
                value: email,
            },
            {
                id: "password",
                value: password,
            }
        ],
    });

    // status is "OK" when successful
    if (signUpResponse.status === "OK") {
        const user = signUpResponse.user;

        // Navigate to home screen
    }
}
```

### Sign In Flow

You can use the `emailPasswordSignIn` function after collecting information from the user.

```tsx
import ThirdPartyEmailPassword from "supertokens-web-js/recipe/thirdpartyemailpassword";

async function signUp(email: string, password: string) {
    const signUpResponse = await ThirdPartyEmailPassword.emailPasswordSignIn({
        formFields: [
            {
                id: "email",
                value: email,
            },
            {
                id: "password",
                value: password,
            }
        ],
    });

    // status is "OK" when successful
    if (signUpResponse.status === "OK") {
        const user = signUpResponse.user;

        // Navigate to home screen
    } else if (signUpResponse.status === "WRONG_CREDENTIALS_ERROR") {
        // Incorrect credentials provided
    }
}
```
