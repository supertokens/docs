### Step 1: Wrap the component in which you want to get the info with `SessionAuth`

```jsx
import { SessionAuth } from 'supertokens-auth-react/recipe/session';

render() {
      return (
            <SessionAuth requireAuth={true}>
                  <Dashboard />
            </SessionAuth>
      );
}
```

### Step 2: This is how to use the session context in a component:

```jsx
import React from "react";
// highlight-next-line
import { useSessionContext } from 'supertokens-auth-react/recipe/session'; 

// Your dashboard component
function Dashboard(props) {
      // highlight-next-line
      let {userId, jwtPayload} = useSessionContext(); 

      let role = jwtPayload.role;

      }
}
```
